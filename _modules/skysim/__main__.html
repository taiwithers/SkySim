 18def parse_args(args: list[str] | None) -> argparse.Namespace:
 19    """Parse command line arguments.
 20
 21    Parameters
 22    ----------
 23    args : list[str] | None
 24        Arguments.
 25
 26    Returns
 27    -------
 28    argparse.Namespace
 29        Parsed arguments.
 30    """
 31    pyproject = read_pyproject()
 32    executable = list(pyproject["scripts"].keys())[0]
 33    version_string = f"{pyproject['name']} {pyproject['version']}"
 34
 35    parser = argparse.ArgumentParser(
 36        prog=executable,
 37        description=pyproject["description"],
 38        epilog="\n".join(
 39            [
 40                version_string,
 41                "Repository: " + pyproject["urls"]["repository"],
 42                "License: " + pyproject["license"],
 43            ]
 44        ),
 45        formatter_class=argparse.RawTextHelpFormatter,
 46    )
 47    parser.add_argument("config_file", help="TOML configuration file")
 48    parser.add_argument(
 49        "--debug", help="print full Python traceback", action="store_true"
 50    )
 51    parser.add_argument(
 52        "--overwrite", help="overwrite existing file(s)", action="store_true"
 53    )
 54    parser.add_argument(
 55        "--verbose",
 56        help="select output verbosity (default %(default)s)",
 57        choices=[0, 1, 2],
 58        type=int,  # casts input to int before applying choices
 59        default=1,
 60    )
 61    parser.add_argument(
 62        "--version",
 63        version=version_string,
 64        action="version",
 65    )
 66
 67    parsed_args = parser.parse_args(args)
 68
 69    return parsed_args

 72def main(  # pylint: disable=inconsistent-return-statements
 73    args: list[str] | None = None,
 74) -> Path | None:
 75    """Entrypoint for the SkySim package. Calls the high-level functions from
 76    the other modules.
 77
 78    Parameters
 79    ----------
 80    args : list[str]|None, optional
 81        When called via command line, this is None, and argparse captures the arguments
 82        directly.
 83        When running `main()` inside of python (e.g., with pytest), `args` holds a list
 84        of arguments.
 85
 86    Returns
 87    -------
 88    Path
 89        Path object to the created file.
 90
 91    Raises
 92    ------
 93    ValueError, ConnectionError
 94        Re-raised from their creation inside the SkySim code if and only if
 95        the --debug flag is set.
 96
 97    SystemExit
 98        Supercedes internal errors to produce a cleaner error message if the
 99        --debug flag is not set.
100    """
101    options = parse_args(args)
102
103    try:
104        config_path = confirm_config_file(options.config_file)
105
106        image_settings, plot_settings = load_from_toml(config_path)
107
108        overwritten_path = check_for_overwrite(plot_settings)  # type: ignore[arg-type]
109        if (overwritten_path is not None) and (not options.overwrite):
110            raise ValueError(
111                "Running SkySim would overwrite one or more files, use the "
112                "--overwrite flag or change/remove the output path to continue."
113            )
114
115        star_table = get_star_table(
116            image_settings.observation_radec,
117            image_settings.field_of_view,
118            image_settings.maximum_magnitude,
119            image_settings.object_colours,
120        )
121
122        body_locations = get_body_locations(
123            image_settings.observation_times, image_settings.earth_location
124        )
125        planet_tables = get_planet_table(body_locations)
126
127        image = create_image_matrix(
128            image_settings,  # type: ignore[arg-type]
129            planet_tables,
130            star_table,
131            options.verbose,
132        )
133
134        create_plot(plot_settings, image, options.verbose)  # type: ignore[arg-type]
135
136    # Optionally print simple error message instead of full traceback
137    except (ValueError, ConnectionError) as e:
138        if options.debug:
139            raise e
140
141        sys.stderr.write(f"skysim: error: {' '.join(e.args)}")
142        sys.exit(1)
143
144    if options.debug:
145        return plot_settings.filename
146    return  # type: ignore[return-value]
